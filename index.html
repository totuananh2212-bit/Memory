/motor-jump-game /images - motorcycle.png - rock.png - waterhole.png - monster.png - index.html - script.js Motor Jump Game const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); canvas.width = 800; canvas.height = 400; const images = { motorcycle: new Image(), rock: new Image(), waterhole: new Image(), monster: new Image(), }; images.motorcycle.src = 'images/motorcycle.png'; images.rock.src = 'images/rock.png'; images.waterhole.src = 'images/waterhole.png'; images.monster.src = 'images/monster.png'; const motorcycle = { x: 100, y: canvas.height - 100, width: 80, height: 40, speed: 5, jumpHeight: 0, isJumping: false, }; const obstacles = []; const monsters = []; let score = 0; let gameSpeed = 3; let isGameOver = false; document.addEventListener('keydown', (e) => { if (e.code === 'Space' && !motorcycle.isJumping) { motorcycle.isJumping = true; motorcycle.jumpHeight = 0; } }); function createObstacle() { const type = Math.random() < 0.5 ? 'rock' : 'waterhole'; const x = canvas.width; const y = type === 'rock' ? canvas.height - 50 : canvas.height - 30; const width = type === 'rock' ? 50 : 60; const height = type === 'rock' ? 50 : 30; obstacles.push({ type, x, y, width, height }); } function createMonster() { const x = canvas.width; const y = Math.random() * (canvas.height - 50); monsters.push({ x, y, width: 50, height: 50 }); } function update() { if (isGameOver) return; ctx.clearRect(0, 0, canvas.width, canvas.height); // Update motorcycle position if (motorcycle.isJumping) { motorcycle.jumpHeight += 2; motorcycle.y -= 2; if (motorcycle.jumpHeight > 50) { motorcycle.isJumping = false; } } else if (motorcycle.y < canvas.height - 100) { motorcycle.y += 2; } // Draw motorcycle ctx.drawImage(images.motorcycle, motorcycle.x, motorcycle.y, motorcycle.width, motorcycle.height); // Update and draw obstacles obstacles.forEach((obstacle, index) => { obstacle.x -= gameSpeed; ctx.drawImage(images[obstacle.type], obstacle.x, obstacle.y, obstacle.width, obstacle.height); if (obstacle.x + obstacle.width < 0) { obstacles.splice(index, 1); score++; if (score % 100 === 0) { gameSpeed++; } } }); // Update and draw monsters monsters.forEach((monster, index) => { monster.x -= gameSpeed; ctx.drawImage(images.monster, monster.x, monster.y, monster.width, monster.height); if (monster.x + monster.width < 0) { monsters.splice(index, 1); } }); // Check for collisions obstacles.forEach((obstacle) => { if ( motorcycle.x + motorcycle.width > obstacle.x && motorcycle.x < obstacle.x + obstacle.width && motorcycle.y + motorcycle.height > obstacle.y && motorcycle.y < obstacle.y + obstacle.height ) { isGameOver = true; } }); monsters.forEach((monster) => { if ( motorcycle.x + motorcycle.width > monster.x && motorcycle.x < monster.x + monster.width && motorcycle.y + motorcycle.height > monster.y && motorcycle.y < monster.y + monster.height ) { isGameOver = true; } }); // Draw score ctx.fillStyle = 'white'; ctx.font = '20px Arial'; ctx.fillText('Điểm: ' + score, 10, 30); if (isGameOver) { ctx.fillStyle = 'black'; ctx.font = '30px Arial'; ctx.fillText('Game Over!', canvas.width / 2 - 80, canvas.height / 2); } requestAnimationFrame(update); } function startGame() { setInterval(() => { if (!isGameOver) { createObstacle(); if (Math.random() < 0.1) { createMonster(); } } }, 2000); update(); } startGame();
